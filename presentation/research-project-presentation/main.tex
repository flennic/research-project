    \section{Introduction}
    \begin{frame}{Introduction | Research Question}
        \begin{itemize}
            \item Syntactic and \alert{S}emantic \alert{D}ependency \alert{P}arsing (SDP)
            \item Acyclic Graphs
            \item Research Question: To what extend does predicted syntax help for semantic dependency parsing?
        \end{itemize}
    \end{frame}

    \section{Overview}
    \begin{frame}{Overview | Objectives}
        \begin{itemize}
            \item Allow to import the data from the \texttt{.conllu} format
            \item Increase the performance during the pre-processing by using caching to save computation time
            \item Implement multitask learning to simultaneously learn syntactic and semantic graphs and investigate the results
        \end{itemize}
    \end{frame}
    
    \section{Objectives}
    \begin{frame}{Objectives | Import \texttt{.conllu}}
        \begin{itemize}
            \item Separate files for syntax (\texttt{.cpn}) and semantic (\texttt{.sdp})
            \item Combined file format \alert{CoNLL-U} from \cite{10.5555/1596276.1596305}

    \end{frame}
